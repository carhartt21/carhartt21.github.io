\section{Open-Source Projects \& Datasets}
% ==============================================================================

\cvitem{AWARE}{Weather-augmentation benchmark framework comprising three pipelines:
  \textbf{SWIFT} (augmentation), \textbf{PRISM} (generation), and \textbf{PROVE} (evaluation).
  \newline\url{https://carhartt21.github.io/AWARE}}

\cvitem{AWACS}{Weather-balanced augmented dataset — 71\,400 images across
  25 augmentation strategies for robustness evaluation.}

\cvitem{SkyCloud360}{600 equirectangular panoramic images for sky and cloud segmentation.
  \newline Code: \url{https://github.com/carhartt21/SkyCloud360}
  \quad Data: \url{https://osf.io/a5ews}}

\cvitem{SkyCloudNet}{Sky and cloud segmentation network with pre-trained models.
  \newline Code: \url{https://github.com/carhartt21/SkyCloudNet}
  \quad Data: \url{https://osf.io/y69ah}}

\cvitem{OUTSIDE15k}{15\,000 outdoor images with 24-class pixel-level annotations.
  \newline Code \& Data: \url{https://github.com/carhartt21/outsideNet}}

\cvitem{360° Detection}{2\,500 panoramic images, 81 traffic sign classes.
  \newline Code: \url{https://github.com/carhartt21/360-Object-Detection}
  \quad Data: \url{https://osf.io/fsteu}}
