\section{Research Profile}
% ==============================================================================
\cvitem{}{%
  \justifying
  My research lies at the intersection of \textbf{computer vision} and
  \textbf{outdoor scene understanding}. I develop multi-scale neural network
  architectures for semantic segmentation of unconstrained outdoor environments,
  encompassing perspectives from standard cameras to 360\textdegree{} panoramic imagery.
  Key contributions include novel datasets, architectures for sky/cloud
  segmentation, traffic sign detection in equirectangular images, and a
  comprehensive weather-augmentation benchmark (AWARE) for evaluating model
  robustness under adverse conditions.%
}

\section{Open-Source Projects \& Datasets}
% ==============================================================================

\cvitem{\textbf{AWARE}}{%
  Comprehensive weather-augmentation benchmark framework for semantic segmentation robustness.
  Three modular pipelines: \textbf{SWIFT} (classical augmentation), \textbf{PRISM} (generative synthesis), and \textbf{PROVE} (evaluation).
  Benchmarks 25 augmentation strategies across 7 weather categories on Cityscapes.
  \newline\faGlobe~\url{https://carhartt21.github.io/AWARE}
  \enspace \ghrepo{carhartt21/SWIFT}
  \enspace \ghrepo{carhartt21/PRISM}
  \enspace \ghrepo{carhartt21/PROVE}}

\cvitem{\textbf{AWACS}}{%
  Weather-balanced Augmented Cityscapes dataset --- \textbf{71\,400 images} generated from Cityscapes val/test splits using
  25 diverse augmentation strategies. Includes pixel-accurate segmentation labels and augmentation metadata.}

\cvitem{\textbf{SkyCloud360}}{%
  \textbf{600} equirectangular panoramic images with pixel-level sky/cloud annotations.
  Paired with a segmentation pipeline addressing projection-induced distortions.
  \newline\ghrepo{carhartt21/SkyCloud360}
  \enspace \faDatabase~\url{https://osf.io/a5ews}}

\cvitem{\textbf{SkyCloudNet}}{%
  Sky and cloud segmentation network for standard perspective images, with pre-trained weights.
  \newline\ghrepo{carhartt21/SkyCloudNet}
  \enspace \faDatabase~\url{https://osf.io/y69ah}
  \enspace \faBrain~\url{https://osf.io/duvkq} (models)}

\cvitem{\textbf{OUTSIDE15k}}{%
  Universal outdoor scene dataset: \textbf{15\,000 images} annotated across \textbf{24 semantic classes}.
  \newline\ghrepo{carhartt21/outsideNet}}

\cvitem{\textbf{360\textdegree{} Detection}}{%
  Traffic sign dataset: \textbf{2\,500 panoramic images} with bounding boxes for \textbf{81 sign classes}.
  Deployed in NOVASIB GmbH's road inventory workflow.
  \newline\ghrepo{carhartt21/360-Object-Detection}
  \enspace \faDatabase~\url{https://osf.io/fsteu}}
